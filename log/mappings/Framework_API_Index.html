// 標準初始化流程 (處理 ProjectItemID 避免斷言失敗) using namespace tracktion_engine; te::Engine engine { "MusiCode" }; auto editState = te::createEmptyEdit(engine); auto id = ProjectItemID::fromProperty(editState, IDs::projectID); if (!id.isValid()) id = ProjectItemID::createNewID(0); te::Edit::Options options { engine }; options.editState = editState; options.editProjectItemID = id; options.role = te::Edit::forEditing; options.numUndoLevelsToStore = 100; auto edit = std::make_unique(options); // 雙重更新同步模式 (Dual Update Pattern) - 封裝於 ParameterDispatcher // 同時處理 AutomatableParameter (音訊執行緒) 與 ValueTree (持久化與狀態同步) MusiCode::ParameterDispatcher::setParameter(targetPlugin, "filterFreq", 0.5f); // 穩定音軌定位 (Stable Track ID Pattern) - 封裝於 TrackManager // 避免使用 trackIndex (0, 1, 2...)，改用 EditItemID 以應對音軌移動或刪除 auto trackID = MusiCode::TrackManager::stringToID("1003"); auto* track = MusiCode::TrackManager::findAudioTrackByID(edit, trackID); // 獲取所有音軌資訊 (供前端渲染同步) juce::var tracksJson = MusiCode::TrackManager::getTracksInfo(edit); // [新增] 模組化傳輸與錄音控制 (Modular Transport & Recording) // 播放與定位 audioEngine.getTransportController().play(false); // false = 不循環 audioEngine.getTransportController().setPosition(tracktion::TimePosition::fromSeconds(2.0)); // 錄音流程 (依照手冊 [3. 播放與傳輸]) if (audioEngine.getRecordingController().isArmed(trackID)) audioEngine.getRecordingController().startRecording(); // 停止並儲存 (生成 Clip) audioEngine.getRecordingController().stopAndSave();
本專案關鍵點：
  • 初始化順序： 變更輸入路由 (setTarget/setRecordingEnabled) 後，必須立即呼叫 ensureContextAllocated()，否則錄音引擎無法識別新的武裝狀態。
  • 錄音核心： 呼叫 transport.record(false) 啟動。停止時需呼叫 stop(false, false) 才能保留 Clip。
  • 武裝狀態： 錄音前必須確保軌道已透過 setRecordingEnabled(true) 開啟武裝，且 MonitorMode 依需求設定。
  • 測試音隔離： 確保 setupTestScene 無殘留 Clip，且全域 MonitorMode 初始設為 OFF。
  • 執行緒規範： 參數異動必須在 Message Thread 執行。ParameterDispatcher 已內建 callAsync 處理。
